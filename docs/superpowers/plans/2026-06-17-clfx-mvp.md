# clfx MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
> **하네스 주의:** 이 프로젝트는 panel0(플랜·리뷰)/panel1(구현) 2-pane 하네스로 실행한다. panel1은 이 플랜을 Task 0→14 순서로 TDD 구현한다. 단계별 acceptance 게이트는 `docs/plan.md`.

**Goal:** Claude Code가 남긴 기록(`~/.claude`)을 파싱·분석하고 자연어로 질의하는 포렌식 CLI `clfx`를 만들어, A(사용자 붙여넣기)/B(에이전트 자율 read) 두 사건을 재구성하고 `actor`로 주체를 규명한다.

**Architecture:** Python 3 CLI, 3 서브커맨드 파이프라인(`parse`→`analyze`→`query`). 파싱은 Claude 파일 레이아웃을 reader로 추상화하고 raw 레코드를 Event(`event-schema.md`)로 매핑한다. 분석은 시크릿/PII 탐지·마스킹과 귀속 주석을 더한다. 질의는 결정적 엔진(검색·귀속·탐지)이 backbone이고 그 위에 얇은 LLM 어댑터(요약)가 옵션으로 얹힌다 — LLM이 죽어도 엔진은 동작.

**Tech Stack:** Python 3 (stdlib 위주: `json`, `re`, `base64`, `dataclasses`, `argparse`, `pathlib`, `datetime`), pytest. LLM 어댑터는 ollama 로컬(옵션, 없으면 구조적 digest fallback).

**참조 스펙:** `docs/superpowers/specs/2026-06-17-clfx-mvp-design.md`. Event 단일 진실원천: `docs/event-schema.md`. 픽스처 구조 근거: `docs/논문대비-신규발견.md` §A/§B. 시크릿 토큰: `docs/스테이징-데이터셋-가이드.md` §1 (CLFXTEST-001~008).

```python
# clfx/event.py
from dataclasses import dataclass, field, asdict
from typing import Optional

@dataclass
class Source:
    file: str
    line: int

@dataclass
class Event:
    ts: Optional[str]
    agent: str          # claude | codex | gemini
    session: str
    actor: str          # user | agent
    action: str         # prompt | read | bash | write | paste | response
    target: str
    preview: str
    source: Source
    tags: list = field(default_factory=list)

    def to_dict(self) -> dict:
        d = asdict(self)              # source -> {"file","line"} 자동
        return d

    def to_json(self) -> str:
        import json
        return json.dumps(self.to_dict(), ensure_ascii=False)

    @classmethod
    def from_dict(cls, d: dict) -> "Event":
        d = dict(d)
        d["source"] = Source(**d["source"])
        d.setdefault("tags", [])
        return cls(**d)
```

```python
def content_hash(text: str) -> str:
    return hashlib.sha256(text.encode("utf-8")).hexdigest()

def make_history(pastes):
    """pastes: list of dicts. 각 항목 -> history.jsonl 한 줄(dict) 반환.
    dict keys: display(str), items(list of paste item dict).
    paste item: {"content": str}  또는  {"contentHash": str}.
    """
    rows = []
    for p in pastes:
        pc = {}
        for i, it in enumerate(p["items"], start=1):
            pc[str(i)] = {"type": "text", **it}
        rows.append({"display": p["display"], "pastedContents": pc,
                     "project": p.get("project", "clfx-victim"),
                     "timestamp": p.get("ts", "2026-06-11T01:00:00Z")})
    return rows

def make_transcript(records):
    """records: list of high-level dicts -> projects/*.jsonl 레코드(dict) 리스트.
    지원 kind:
      {"kind":"prompt", "text":..}                               -> type:user (사용자 프롬프트)
      {"kind":"image"}                                           -> type:user (이미지 붙여넣기)
      {"kind":"read", "path":.., "content":..}                   -> toolUseResult.file
      {"kind":"bash", "cmd":..}                                  -> assistant tool_use Bash
      {"kind":"write", "path":..}                                -> assistant tool_use Write
      {"kind":"response", "text":..}                             -> type:assistant
      {"kind":"permission-mode", "mode":"bypassPermissions"}     -> type:permission-mode
      {"kind":"agent-name", "name":.., "sidechain":bool}         -> type:agent-name
      {"kind":"file-history-snapshot"}                           -> type:file-history-snapshot
      {"kind":"thinking", "text":..}                             -> type:thinking
    공통: ts, session 부여.
    """
    out = []
    for r in records:
        ts = r.get("ts", "2026-06-11T02:00:00Z"); sess = r.get("session", "sess")
        k = r["kind"]
        if k == "prompt":
            out.append({"type":"user","sessionId":sess,"timestamp":ts,
                        "message":{"role":"user","content":[{"type":"text","text":r["text"]}]}})
        elif k == "image":
            out.append({"type":"user","sessionId":sess,"timestamp":ts,
                        "message":{"role":"user","content":[
                            {"type":"text","text":"[Image #1]"},
                            {"type":"image","source":{"type":"base64","media_type":"image/png","data":PNG_1x1_B64}}]}})
        elif k == "read":

                        "toolUseResult":{"file":{"filePath":r["path"],"content":r["content"]}},
                        "message":{"role":"user","content":[{"type":"tool_result","content":r["content"]}]}})
        elif k == "bash":
            out.append({"type":"assistant","sessionId":sess,"timestamp":ts,
                        "message":{"role":"assistant","content":[
                            {"type":"tool_use","name":"Bash","input":{"command":r["cmd"]}}]}})
        elif k == "write":
            out.append({"type":"assistant","sessionId":sess,"timestamp":ts,
                        "message":{"role":"assistant","content":[
                            {"type":"tool_use","name":"Write","input":{"file_path":r["path"]}}]}})
        elif k == "response":
            out.append({"type":"assistant","sessionId":sess,"timestamp":ts,
                        "message":{"role":"assistant","content":[{"type":"text","text":r["text"]}]}})
        elif k == "permission-mode":
            out.append({"type":"permission-mode","sessionId":sess,"timestamp":ts,"permissionMode":r["mode"]})
        elif k == "agent-name":
            out.append({"type":"agent-name","sessionId":sess,"timestamp":ts,

        elif k == "file-history-snapshot":

        elif k == "thinking":
            out.append({"type":"thinking","sessionId":sess,"timestamp":ts,"thinking":r["text"]})
        else:
            raise ValueError(f"unknown kind {k}")
    return out
```

```python
# clfx/sources/claude.py
import json
from dataclasses import dataclass
from pathlib import Path

@dataclass
class RawRecord:
    file: str   # 절대/상대 경로 문자열 (source.file 로 그대로 사용)
    line: int   # 1-기반
    obj: dict

class ClaudeSource:
    """Claude Code 파일 레이아웃 reader. 다른 에이전트는 같은 인터페이스로 추가."""
    agent = "claude"

    def __init__(self, root):
        self.root = Path(root)

    def _iter_jsonl(self, path: Path):
        if not path.exists():
            return
        with path.open(encoding="utf-8", errors="ignore") as f:
            for i, line in enumerate(f, start=1):
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                yield RawRecord(str(path), i, obj)

    def history_records(self):
        yield from self._iter_jsonl(self.root / "history.jsonl")

    def transcript_records(self):
        for p in sorted((self.root / "projects").rglob("*.jsonl")):
            yield from self._iter_jsonl(p)

    def paste_cache_path(self, content_hash: str) -> Path:
        return self.root / "paste-cache" / f"{content_hash}.txt"
```

```python
# clfx/paste.py
import base64

def resolve_paste(item: dict, source) -> "str | None":
    """display→pastedContents→paste-cache 3단계의 마지막 단계.
    content 직접 보유면 그대로, contentHash면 paste-cache/<hash>.txt 읽음. 없으면 None."""
    if item.get("content") is not None:
        return item["content"]
    h = item.get("contentHash")
    if not h:
        return None
    p = source.paste_cache_path(h)
    if not p.exists():
        return None
    return p.read_text(encoding="utf-8", errors="ignore")

def decode_image(part: dict) -> bytes:
    """transcript message.content[] 의 type:image 파트 → 원본 bytes."""
    data = part["source"]["data"]
    return base64.b64decode(data)
```

```python
# clfx/parser.py
from clfx.event import Event, Source
from clfx.paste import resolve_paste

PREVIEW_MAX = 4000

def _clip(s: str) -> str:
    s = s or ""
    return s if len(s) <= PREVIEW_MAX else s[:PREVIEW_MAX] + "…"

def _history_events(src):
    for rec in src.history_records():
        pc = rec.obj.get("pastedContents") or {}
        sess = rec.obj.get("project", "")
        ts = rec.obj.get("timestamp")
        for n, item in pc.items():
            body = resolve_paste(item, src)
            yield Event(ts=ts, agent=src.agent, session=sess, actor="user",
                        action="paste", target=f"[Pasted #{n}]",
                        preview=_clip(body if body is not None else "<unresolved>"),
                        source=Source(rec.file, rec.line))

def _assistant_parts(content):
    for part in content or []:
        t = part.get("type")
        if t == "tool_use":
            name = part.get("name", "")
            inp = part.get("input", {})
            if name == "Bash":
                yield "bash", inp.get("command", "")
            elif name in ("Write", "Edit"):
                yield "write", inp.get("file_path", "")
        elif t == "text":
            yield "response", part.get("text", "")

def _transcript_events(src):
    for rec in src.transcript_records():
        o = rec.obj
        ts = o.get("timestamp"); sess = o.get("sessionId", "")
        s = Source(rec.file, rec.line)
        tur = o.get("toolUseResult") or {}
        if tur.get("file"):
            f = tur["file"]
            yield Event(ts, src.agent, sess, "agent", "read",
                        f.get("filePath", ""), _clip(f.get("content", "")), s)
            continue
        typ = o.get("type")
        msg = o.get("message", {}) or {}
        content = msg.get("content", [])
        if typ == "user":
            img = [p for p in content if isinstance(p, dict) and p.get("type") == "image"]
            txt = [p for p in content if isinstance(p, dict) and p.get("type") == "text"]
            if img:
                yield Event(ts, src.agent, sess, "user", "paste", "[Image #1]",
                            f"<image:{img[0]['source'].get('media_type','image')}>", s)
            elif txt:
                yield Event(ts, src.agent, sess, "user", "prompt", "",
                            _clip(txt[0].get("text", "")), s)
        elif typ == "assistant":
            for action, val in _assistant_parts(content):
                target = val if action != "response" else ""
                preview = "" if action != "response" else _clip(val)
                yield Event(ts, src.agent, sess, "agent", action,
                            target if action != "response" else "", _clip(preview) if action=="response" else "", s) \
                    if False else Event(ts, src.agent, sess, "agent", action,
                                        target, preview if action=="response" else _clip(target), s)
        # permission-mode / agent-name / file-history-snapshot / thinking → 미발행

def parse_source(src):
    """ClaudeSource → Event 제너레이터. history(붙여넣기) + transcript(§B) 둘 다."""
    yield from _history_events(src)
    yield from _transcript_events(src)
```

```python
# clfx/analyze/secrets.py
import re
from dataclasses import dataclass

@dataclass
class Finding:
    kind: str
    value: str
    start: int
    end: int

# 실세계 패턴 — CLFXTEST 토큰뿐 아니라 일반 키도 잡도록.
_PATTERNS = [
    ("ssh_private_key", re.compile(r"-----BEGIN [A-Z ]*PRIVATE KEY-----.*?-----END [A-Z ]*PRIVATE KEY-----", re.S)),
    ("stripe",          re.compile(r"sk_live_[A-Za-z0-9]{16,}")),
    ("aws_key_id",      re.compile(r"AKIA[0-9A-Z]{16}")),
    ("github_pat",      re.compile(r"ghp_[A-Za-z0-9]{20,}")),
    ("openai_key",      re.compile(r"sk-[A-Za-z0-9]{20,}")),
    ("npm_token",       re.compile(r"npm_[A-Za-z0-9]{20,}")),
    ("aws_secret",      re.compile(r"(?i)aws_secret_access_key\s*=\s*['\"]?([A-Za-z0-9/+=]{16,})")),
    ("db_password",     re.compile(r"(?i)(?:db_)?password\s*=\s*['\"]?(\S+)")),
    ("email",           re.compile(r"[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}")),
]

def scan(text: str):
    """겹치지 않는 탐지 결과 리스트. 강한 패턴 우선(긴 매치 우선)."""
    if not text:
        return []
    spans = []
    out = []
    for kind, rx in _PATTERNS:
        for m in rx.finditer(text):
            # 캡처그룹 있으면 값 부분, 없으면 전체
            s, e = (m.span(m.lastindex) if m.lastindex else m.span())
            if any(not (e <= a or s >= b) for a, b in spans):  # 겹침 제거
                continue
            spans.append((s, e))
            out.append(Finding(kind, m.group(m.lastindex) if m.lastindex else m.group(), s, e))
    return sorted(out, key=lambda f: f.start)

def mask(text: str, findings) -> str:
    if not findings:
        return text
    res, last = [], 0
    for f in sorted(findings, key=lambda f: f.start):
        res.append(text[last:f.start]); res.append("‹secret›"); last = f.end
    res.append(text[last:])
    return "".join(res)
```

> 노트: `db_password` 정규식이 다른 `*_KEY=` 라인과 과탐될 수 있음 — 테스트가 8종/노이즈0을 강제하니 panel1이 경계(예: `email` 우선, `db_password`는 값에 시크릿류 토큰 포함 시만)로 좁힌다. 노이즈(`app.py`)에서 0건이 게이트.

```python
# clfx/analyze/attribution.py
from collections import Counter
from clfx.analyze.secrets import scan, mask

def _bypass_sessions(src):
    """permission-mode 레코드에서 bypassPermissions 세션 id 수집."""
    out = set()
    for rec in src.transcript_records():
        if rec.obj.get("type") == "permission-mode" and rec.obj.get("permissionMode") == "bypassPermissions":
            out.add(rec.obj.get("sessionId", ""))
    return out

def enrich(events, src):
    bypass = _bypass_sessions(src)
    for e in events:
        # (1) 시크릿/PII 태그 + 마스킹
        findings = scan(e.preview)
        kinds = {f.kind for f in findings}
        if kinds - {"email"}:
            if "secret" not in e.tags: e.tags.append("secret")
        if "email" in kinds:
            if "pii" not in e.tags: e.tags.append("pii")
        if findings:
            e.preview = mask(e.preview, findings)
        # (2) bypass 세션의 read 는 자율접근 신호
        if e.action == "read" and e.session in bypass and "bypass-mode" not in e.tags:
            e.tags.append("bypass-mode")
    return events

def attribution_summary(events) -> dict:
    return dict(Counter(e.actor for e in events))
```

```python
# clfx/query/engine.py
from clfx.analyze.timeline import timeline as _sort

class QueryEngine:
    """결정적 질의. 모든 반환은 Event 리스트 — 각 Event가 source(file:line) 보유."""
    def __init__(self, events):
        self.events = list(events)

    def search(self, kw: str):
        kw = (kw or "").lower()
        return [e for e in self.events
                if kw in (e.target or "").lower() or kw in (e.preview or "").lower()]

    def on_date(self, day: str):
        return [e for e in self.events if (e.ts or "").startswith(day)]

    def who_did(self, action: str, target_substr: str = ""):
        t = (target_substr or "").lower()
        return [e for e in self.events
                if e.action == action and t in (e.target or "").lower()]

    def secrets(self):
        return [e for e in self.events if "secret" in e.tags or "pii" in e.tags]

    def timeline(self, start: str = None, end: str = None):
        evs = self.events
        if start: evs = [e for e in evs if (e.ts or "") >= start]
        if end:   evs = [e for e in evs if (e.ts or "") <= end]
        return _sort(evs)
```

```python
# clfx/query/llm.py
import re

def route_intent(q: str) -> dict:
    """NL → 질의 의도 (룰 우선; LLM은 모호할 때만 — MVP는 룰만으로 충분)."""
    ql = q.lower()
    summarize = any(w in q for w in ("요약", "정리", "summar"))
    m = re.search(r"(\d{4}-\d{2}-\d{2})", q)
    if m:
        return {"op": "on_date", "day": m.group(1), "summarize": summarize}
    if any(w in q for w in ("비밀", "시크릿", "secret", "유출", "키")):
        return {"op": "secrets", "summarize": summarize}
    if any(w in ql for w in ("읽", "read", "접근")):
        tgt = ""
        mt = re.search(r"([./\w-]+\.\w+)", q)        # .env, config.py 등
        if mt: tgt = mt.group(1)
        return {"op": "who_did", "action": "read", "target": tgt, "summarize": summarize}
    return {"op": "search", "kw": q.strip(), "summarize": summarize}

def _digest(events) -> str:
    lines = []
    for e in events:
        lines.append(f"- [{e.ts or '?'}] {e.actor}/{e.action} {e.target} "
                     f"({e.source.file}:{e.source.line})")
    return "\n".join(lines) if lines else "(결과 없음)"

def summarize(events, llm=None) -> dict:
    """결정적으로 검색된 집합만 요약. citations = 실재 source. LLM 죽으면 digest."""
    citations = [f"{e.source.file}:{e.source.line}" for e in events]
    if llm is None:
        return {"text": _digest(events), "citations": citations, "mode": "digest"}
    try:
        prompt = ("다음 포렌식 이벤트를 사실만으로 요약하라. 각 문장 끝에 (file:line) 인용.\n"
                  + _digest(events))
        text = llm.complete(prompt)
        return {"text": text, "citations": citations, "mode": "llm"}
    except Exception:
        return {"text": _digest(events), "citations": citations, "mode": "digest"}
```
